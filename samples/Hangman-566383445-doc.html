    set [GAME OVER? v] to [yes]
    broadcast [you win v] and wait
  end

define pick word
if <(length of [USED WORDS LIST v]) = (length of [WORDS LIST v])> then
  delete all of [USED WORDS LIST v]
end
set [SECRET WORD v] to (item [ANY] of [WORDS LIST v])
repeat until <not <[USED WORDS LIST v] contains [SECRET WORD]?>>
  set [SECRET WORD v] to (item [ANY] of [WORDS LIST v])
end
add [SECRET WORD] to [USED WORDS LIST v]

  change [index v] by [1]
