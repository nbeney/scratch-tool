when this sprite clicked
if <(FACEUP INDEX) = [nil]> then
  set [FACEUP INDEX v] to (index)
  set [FACEUP COSTUME v] to (costume)
  switch costume to (costume)
else
  if <not <(index) = (FACEUP INDEX)>> then
    change [TRIES v] by [1]
    switch costume to (costume)
    wait [0.5] seconds
    if <(costume) = (FACEUP COSTUME)> then
      broadcast [good guess v] and wait
      set [FACEUP INDEX v] to [nil]
      set [FACEUP COSTUME v] to [nil]
      delete this clone

      broadcast [bad guess v] and wait
      set [FACEUP INDEX v] to [nil]
      set [FACEUP COSTUME v] to [nil]
      switch costume to [back v]
    end
  end
end

define initialize arrangement
delete all of [ARRANGEMENT v]
set [costume v] to [2]
repeat until <(length of [ARRANGEMENT v]) = ([2] * (NUM PAIRS))>
  add (costume) to [ARRANGEMENT v]
  add (costume) to [ARRANGEMENT v]
  change [costume v] by [1]
end

define shuffle arrangement
repeat ((length of [ARRANGEMENT v]) * [0.75])
  set [SHUFFLE I v] to (pick random [1] to (length of [ARRANGEMENT v]))
  set [SHUFFLE J v] to (pick random [1] to (length of [ARRANGEMENT v]))
  set [SHUFFLE TEMP v] to (item (SHUFFLE I) of [ARRANGEMENT v])
  replace item (SHUFFLE I) of [ARRANGEMENT v] with (item (SHUFFLE J) of [ARRANGEMENT v])
  replace item (SHUFFLE J) of [ARRANGEMENT v] with (SHUFFLE TEMP)
end
